#include <iostream>
using namespace std;

int main()
{
    int a, n;
    cout << "Enter a number and  and an exponent"<<endl;
    cin >> a >> n;
    int result = 1;
    if (n==0)
    {
        result = 1;
    }
    else
    {
        for (int i = 1; i <= n; i++)
        {

            result = result * a;
        }
    }
    cout << result;
}
